BinaryNode * search(BinaryNode * root, int key) 
{
    if (root == NULL || root->data == key) {
        return root;
    }

    BinaryNode * found = search(root->left, key);
    if (found == NULL) {
        found = search(root->right, key);
    }

    return found;
}
