int searchHashTable(HashTable * hashTable, int key) 
{
    int hashIndex = hashFunction(key);
    HashItem * temp = hashTable->table[hashIndex];

    while (temp != NULL) {
        if (temp->key == key) {
            return temp->value;
        }
        temp = temp->next;
    }

    return -1;
}
