\begin{pseudocode}[caption={Algoritmo de ordenação por contagem}, label={countingSortP}]
COUNTING-SORT(A, n, K)
let B[1:n] and C[0:K] be new arrays
for i $\gets$ 0 to k
    C[i] $\gets$ 0
for j $\gets$ 1 to n
    C[A[j]] $\gets$ C[A[j]] + 1
for i $\gets$ 1 to k
    C[i] = C[i] + C[i - 1]
for j $\gets$ n downto 1
    B[C[A[j]]] $\gets$ A[j]
    C[A[j]] $\gets$ C[A[j]] - 1
return B
\end{pseudocode}
